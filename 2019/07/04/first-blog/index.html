关于springmvc和mybatis的配置文件

web.xml中的配置：

   <?xml version="1.0" encoding="UTF-8"?>
<web-app xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns="http://java.sun.com/xml/ns/javaee" xmlns:web="http://java.sun.com/xml/ns/javaee/web-app_2_5.xsd" xsi:schemaLocation="http://java.sun.com/xml/ns/javaee http://java.sun.com/xml/ns/javaee/web-app_3_0.xsd" id="WebApp_ID" version="3.0">
 <filter>
<filter-name>characterEncodingFilter&lt;/filter-name> 
  <filter-class>org.springframework.web.filter.CharacterEncodingFilter&lt;/filter-class>
  <init-param>
  <param-name>encoding&lt;/param-name>
  <param-value>utf-8&lt;/param-value>
  </init-param>
  </filter>
 <filter-mapping>
   <filter-name>characterEncodingFilter&lt;/filter-name>
   <url-pattern>*.do&lt;/url-pattern>
   </filter-mapping>
 <servlet>
 <servlet-name>demo&lt;/servlet-name>
 <servlet-class>org.springframework.web.servlet.DispatcherServlet&lt;/servlet-class>
  </servlet>
  <servlet-mapping>
<servlet-name>demo&lt;/servlet-name>
 <url-pattern>*.do&lt;/url-pattern>
  </servlet-mapping>
</web-app>

springmvc-servlet.xml文件配置

<?xml version="1.0" encoding="UTF-8"?>
<beans xmlns="http://www.springframework.org/schema/beans"

</beans>

mybatis.xml配置文件

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE configuration
  PUBLIC "-//mybatis.org//DTD Config 3.0//EN"
  "http://mybatis.org/dtd/mybatis-3-config.dtd">
<configuration>
<environments default="default">
<environment id="default">
<transactionManager type="JDBC"/>
<dataSource type="POOLED">
  <property name="driver" value="com.mysql.jdbc.Driver"/>
  <property name="url" value="jdbc:mysql://localhost:3306/student?characterEncoding=utf-8"/>
  <property name="username" value="root"/>
  <property name="password" value="123456"/>
</dataSource>
</environment>
</environments>
<mappers>
<mapper resource="com/hezi/mapper/mapper.xml"/>
</mappers>
</configuration>

com.hezi.mapper下的映射文件mapper.xml

<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE mapper PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN" "http://mybatis.org/dtd/mybatis-3-mapper.dtd">
<mapper namespace="com.hezi.mapper">
<select id="sel" resultType="com.hezi.pojo.User">
select * from user 
</select>
<select id="sel1" resultType="com.hezi.pojo.U" parameterType="com.hezi.pojo.U">
select uname,upass from u where uname=#{uname} and upass=#{upass}
</select>
<insert id="ins" parameterType="com.hezi.pojo.U">
insert into u values(default,#{uname},#{upass})
</insert>
</mapper>
